** Speedy startup

   Temporarily disable garbage collection during
   initialization. Increases the ~gc-cons-threshold~ and restores the default after initialization.

   #+begin_src emacs-lisp :tangle early-init.el

   (let ((file-name-handler-alist-old file-name-handler-alist))
     (add-hook 'emacs-startup-hook
               (lambda ()
                 (setq gc-cons-threshold (* 1024 1024 20))
                 (setq file-name-handler-alist file-name-handler-alist-old)))
     (setq gc-cons-threshold most-positive-fixnum)
     (setq file-name-handler-alist nil))

   (setq inhibit-default-init t)

   #+end_src

*** Company Auctex -> Capf
#+begin_src emacs_lisp

(use-package company-auctex
      :after tex
      :commands company-auctex-macros
                company-auctex-symbols
                company-auctex-environments)

    (defun my/latex-capf-setup ()
      "CAPF setup for LaTeX."
      (setq-local completion-at-point-functions
                  (list
                   ;; LSP completion from Eglot/Digestif
                   #'eglot-completion-at-point
                   ;; AUCTeX's own completion first
                   #'TeX-completion-at-point
                   ;; Reuse company-auctex backends through cape
                   (cape-company-to-capf #'company-auctex-macros)
                   (cape-company-to-capf #'company-auctex-symbols)
                   (cape-company-to-capf #'company-auctex-environments)
                   ;; Generic fallbacks
                   #'cape-file
                   #'cape-dabbrev)))

    (add-hook 'LaTeX-mode-hook #'my/latex-capf-setup)
  
#+end_src
